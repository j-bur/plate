## Usage

### Plugins

```tsx
import { createKbdPlugin } from '@udecode/plate-kbd';

const plugins = createPlugins(
  [
    // ...otherPlugins,
    createKbdPlugin(),
  ],
  {
    components: createPlateUI(),
  }
);
```
