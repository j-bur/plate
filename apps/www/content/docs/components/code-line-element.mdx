## Usage

```tsx
import { createCodeBlockPlugin } from '@udecode/plate-code-block';
import {
  codeLineElement,
} from '@/components/plate-ui/code-line-element';
```

```tsx
createPlugins([
  createCodeBlockPlugin(),
], {
  components: {
    [ELEMENT_CODE_LINE]: CodeLineElement,
  }
})
```
