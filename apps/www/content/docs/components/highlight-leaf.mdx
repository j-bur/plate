## Usage

### Plugins

```tsx
import { createHighlightPlugin } from '@udecode/plate-highlight';

const plugins = createPlugins(
  [
    // ...otherPlugins,
    createHighlightPlugin(),
  ],
  {
    components: createPlateUI(),
  }
);
```
