## Usage

### Plugins

```tsx
import { createPlugins } from '@udecode/createPlugins';
import { createPlateUI } from '@udecode/createPlateUI';
import { ELEMENT_BLOCKQUOTE } from '@udecode/plate-block-quote';
import { BlockquoteElement } from '@udecode/blockquote-element';

createPlugins([createPlateUI()], {
  components: {
    [ELEMENT_BLOCKQUOTE]: BlockquoteElement,
  },
});
```
