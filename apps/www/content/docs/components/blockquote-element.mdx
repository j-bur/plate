## Usage

```tsx
import { createBlockquotePlugin } from '@udecode/plate-block-quote';
import {
  blockquote,
} from '@/components/plate-ui/blockquote-element;
```

```tsx
createPlugins([
  createBlockquotePlugin(),
], {
  components: {
    [ELEMENT_BLOCKQUOTE]: blockquote,
  }
})
```
